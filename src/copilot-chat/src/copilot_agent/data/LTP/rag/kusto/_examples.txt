# Few-shot Examples for Generating KQL Queries in a RAG System
Below are few-shot examples derived from the provided KQL samples. These examples are designed to help a Retrieval-Augmented Generation (RAG) system generate new KQL queries based on user questions.

## Example 1: Weekly Cluster Node Health Summary
Question: "Can you provide a weekly summary of node health states (average availability ratio, allocated VMs, empty VMs, unallocatable VMs...) for specific Cluster (ClusterName)?"

Generated KQL:
```kql
cluster('{cluster_url}').database('{database_name}').table('PBICacheClusterCapacityStatus')
| summarize arg_max(BackupTimeStamp, *) by WeekNum, Cluster
| project-away BackupTimeStamp
```

## Example 2: Daily Node Status Count
Question: "How can I calculate the daily count of node statuses grouped by their attributes?"
- Alias: "What is minimum, average and maximum daily availability ratio for endpoint pai-wcu, in week 30?"

Generated KQL:
```kql
// Step 0: get source data
let NodeStatusRecord = cluster('{cluster_url}').database('{database_name}').table('NodeStatusRecord');
let NodeStatusAttributes = cluster('{cluster_url}').database('{database_name}').table('NodeStatusAttributes');
let VMInfo = cluster('{cluster_url}').database('{database_name}').table('VMInfo');
let ClusterVMSSInfo = cluster('{cluster_url}').database('{database_name}').table('ClusterVMSSInfo');
let min_day = toscalar(NodeStatusRecord | summarize startofday(min(Timestamp)));
let max_day = toscalar(NodeStatusRecord | summarize startofday(max(Timestamp)));
let AllDays = range Day from min_day to max_day step 1d;
// Step 1: Get all Groups from StatusAttributes
let AllGroups = NodeStatusAttributes | summarize by Group;
// Step 2: Build all (Day, Group) combinations
let AllCombos =
AllGroups
| extend dummy = 1
| join kind=inner (
AllDays | extend dummy = 1
) on dummy
| project Day, Group;
// Step 3: Get latest status per node per day
let LatestPerNodePerDay =
NodeStatusRecord
| join kind=leftouter(
VMInfo
) on HostName
| extend VMSS_clean = trim(" ", tolower(VMSS))
| join kind=leftouter(
ClusterVMSSInfo
| extend VMSS_clean = trim(" ", tolower(VMSSName))
) on VMSS_clean
| extend DayList = range(min_day, max_day, 1d)
| mv-expand Day = DayList to typeof(datetime)
| where startofday(Timestamp) <= Day
| summarize arg_max(Timestamp, Status) by HostName, Day, Endpoint, ClusterName;
// Step 4: Map statuses to groups
let NodeStatusWithGroup =
LatestPerNodePerDay
| extend Status_clean = trim(" ", tolower(Status))
| join kind=leftouter (
NodeStatusAttributes
| extend Status_clean = trim_start(@"[^\w]+",Status)
) on Status_clean;
// Step 5: Count nodes per group per day
let DailyGroupCounts =
NodeStatusWithGroup
| summarize NodeCount = count() by Day, Group, Endpoint, ClusterName;
// Step 6: Merge with full (Day, Group) grid to include 0s
let DailyGroupNodeCounts =
AllCombos
| join kind=leftouter (DailyGroupCounts) on Day, Group
| extend NodeCount = coalesce(NodeCount, 0)
| sort by Day asc, Group asc
| project-away Day1, Group1;
// Step 7: calculate daily availability ratio
let DailyAvailabilityRatio = DailyGroupNodeCounts
| summarize arg_max(NodeCount, *) by Day, Group, Endpoint // Ensure all columns are available for pivoting
| evaluate pivot(Group, sum(NodeCount))
| where isnotempty(Endpoint) // Add this line to filter out empty Endpoints
| order by Day asc, Endpoint asc
| extend Week = weekofyear(startofweek(Day - 1d, 0)) // Monday as the start of the week
| extend HealthyCount = Healthy
| extend TotalCount = Cordon + Healthy + New + UA + Validation
| extend DailyAvailablityRatio = iff(TotalCount > 0, toreal(HealthyCount) / toreal(TotalCount), 0.0);
// Step 8: summarize weekly availability ratio
let Res = DailyAvailabilityRatio
| summarize 
AverageDailyAvailabilityRatio = avg(DailyAvailablityRatio),
MinimumDailyAvailabilityRatio = min(DailyAvailablityRatio),
MaximumDailyAvailabilityRatio = max(DailyAvailablityRatio)
by Week, Endpoint, ClusterName;
Res
```

## Example 6: Node Action Flow and Node MTBF Analysis
Question: "How can I analyze the sequence of node actions related to hardware failures and calculate the mean time between failures (MTBF)?"

Generated KQL:
```kql
// preserve the actions to filter, no reference
let States = datatable (State:string)
[
    "new-validating", "validating-available", "available-cordoned", "cordoned-triaged_hardware"
];
// step 1, filter the actions related to Hardware failure, assign an action seq, join to extend the Endpoint and ClusterName column
let AllRelevantActions = cluster('{cluster_url}').database('{database_name}').table('NodeActionRecord')
| where Action in ("new-validating", "validating-available", "available-cordoned", "cordoned-triaged_hardware") // Filter for relevant actions to optimize performance
| extend ActionSeq = case(
    Action == "new-validating", "S00",
    Action == "validating-available", "S01",
    Action == "available-cordoned", "S02",
    Action == "cordoned-triaged_hardware", "S03",
    "S99"
)
| join kind=leftouter 
    (VMInfo 
    | distinct HostName, VMSS)
on HostName
    | extend VMSS_clean = trim(" ", tolower(VMSS))
    | join kind=leftouter(
        ClusterVMSSInfo
        | extend VMSS_clean = trim(" ", tolower(VMSSName))
    ) on VMSS_clean
| project-away Reason, Detail, Category
| project HostName, Timestamp, Action, ActionSeq, Endpoint, ClusterName
| sort by HostName asc, Timestamp asc, ActionSeq asc; // Sort by HostName and Timestamp for correct sequence matching; Sort by ActionSeq is necessary as some actions could be registered to the same timestamp
// step 2, create a action flow pattern
let TransactionTable = AllRelevantActions
| sort by HostName asc, Timestamp asc, ActionSeq asc
| serialize
| extend preAction = prev(Action, 1)
| extend nextAction = next(Action, 1)
| extend preHostName = prev(HostName, 1)
| extend nextHostName = next(HostName, 1)
| extend isBackVA = iff(Action == "validating-available" and (HostName == preHostName and (preAction == "new-validating" or preAction == "cordoned-triaged_hardware")), 1, 0)
| extend isNewAC = iff(Action == "available-cordoned" and (HostName == nextHostName and nextAction == "cordoned-triaged_hardware"), 1, 0)
| extend isNewCTH = iff(Action == "cordoned-triaged_hardware" and (HostName == preHostName and preAction == "available-cordoned"), 1, 0);
// step 3, filter out the actions using above pattern
let FocusedTransaction = TransactionTable
| where isNewAC == 1 or isNewCTH == 1 or isBackVA == 1; // keep only entries with state transactions we desire
// step 4, calculate duration between failures for the same hostname
let FaiureDuration = FocusedTransaction
| serialize // This operator is required to use row-level functions like 'next()'
| extend
    LastAction = prev(Action, 1), // Get the action of the next record
    LastTimestamp = prev(Timestamp, 1), // Get the timestamp of the next record
    LastHostName = prev(HostName, 1) // Get the HostName of the next record
| where
    Action == "available-cordoned" // The current action must be 'available-cordoned'
    and LastAction == "validating-available" // The next action must be 'cordoned-triaged_hardware'
    and HostName == LastHostName // Both actions must belong to the same HostName
| project
    HostName,
    Endpoint,
    ClusterName,
    AvailableCordonedTimestamp = Timestamp, // Rename for clarity
    ValidatingAvailableTimestamp = LastTimestamp // Rename for clarity
| extend DurationBetweenFailureSpan = AvailableCordonedTimestamp - ValidatingAvailableTimestamp
| order by HostName asc, AvailableCordonedTimestamp asc // Order the final results for better readability
| extend Week = weekofyear(startofweek(AvailableCordonedTimestamp - 1d, 0)) // Monday as the start of the week
| extend DurationBetweenFailureDay = toreal(DurationBetweenFailureSpan) / 36000000000 /24
| extend DurationBetweenFailureHour = toreal(DurationBetweenFailureSpan) / 36000000000;
// step 5, average aggregate
let Res = FaiureDuration
| order by Week asc
| summarize 
    MTBFSpan = avg(DurationBetweenFailureSpan),
    MTBFDays = avg(DurationBetweenFailureDay),
    MTBFHours = avg(DurationBetweenFailureHour),
    _debugTotalAvailableDays = sum(DurationBetweenFailureDay),
    SampleCounts = count()
by Week, Endpoint, ClusterName
| order by Endpoint asc, ClusterName asc, Week asc;
let CumulativeRes = Res
| extend CumulativeTotalAvaiDays = row_cumsum(SampleCounts * MTBFDays, Endpoint != prev(Endpoint) or ClusterName != prev(ClusterName))
| extend CumulativeTotalAvaiHours = row_cumsum(SampleCounts * MTBFHours, Endpoint != prev(Endpoint) or ClusterName != prev(ClusterName))
| extend CumulativeSampleCount = row_cumsum(SampleCounts, Endpoint != prev(Endpoint) or ClusterName != prev(ClusterName))
| extend CumulativeMTBFDays = iff(CumulativeSampleCount > 0, todouble(CumulativeTotalAvaiDays) / CumulativeSampleCount, 0.0)
| extend CumulativeMTBFHours = iff(CumulativeSampleCount > 0, todouble(CumulativeTotalAvaiHours) / CumulativeSampleCount, 0.0);
CumulativeRes
```

## Example 7: OFR (out for repair) Node Recycle Time Analysis
Question: "How can I calculate the weekly node recycle time for OFR (out for repair) nodes?"

Generated KQL:
```kql
// Step 1: Prepares the Data and Adds Flags to Categorize Each Row's Status
let start_actions = dynamic(["cordoned", "new"]);
let ua = "ua";
let ua_dealloc = "deallocated_ua";
let ua_alloc = "allocated_ua";
let end_actions = dynamic(["available", "deallocated_capacity"]);
let data_with_flags = cluster('{cluster_url}').database('{database_name}').NodeStatusRecord
| distinct Timestamp, HostName, Status, Endpoint
| join kind=leftouter 
    (VMInfo 
    | distinct HostName, VMSS)
on HostName
    | extend VMSS_clean = trim(" ", tolower(VMSS))
    | join kind=leftouter(
        ClusterVMSSInfo
        | extend VMSS_clean = trim(" ", tolower(VMSSName))
    ) on VMSS_clean
| extend Timestamp = todatetime(['Timestamp'])
| where isnotempty(HostName) and isnotempty(Status)
| sort by HostName  asc, Timestamp  asc, Status asc
| serialize
| extend
    is_start = iif(Status in (start_actions), 1, 0),
    is_end = iif(Status in (end_actions), 1, 0),
    is_ua = iif(Status == ua, 1, 0),
    is_ua_dealloc = iif(Status == ua_dealloc, 1, 0),
    is_ua_alloc = iif(Status == ua_alloc, 1, 0)
| extend is_big_start = iff(Status in (start_actions) and prev(Status) in (end_actions), 1 ,0)
| extend big_group_id = row_cumsum(is_big_start)
| extend group_key = strcat(HostName, "_", tostring(big_group_id))
| serialize
| extend sub_group_id = row_cumsum(is_start, group_key != prev(group_key))
| project-away HostName1;
// Step 2: Handles Duplicates and Finalizes Data
let duplicated_rows =
    data_with_flags
    | where is_start == 1 and sub_group_id >1
    | extend sub_group_id = sub_group_id - 1;
let events_by_subgroup =
    data_with_flags
    | union duplicated_rows
    | sort by HostName, Timestamp, sub_group_id;
// Step 3: Summarizes and Calculates Time Metrics per Sub-Group
let subgroup_metrics =
events_by_subgroup
| summarize
    StartTime = minif(Timestamp, is_start == 1),
    UAedTime = minif(Timestamp, is_ua == 1),
    UADeallocTime = minif(Timestamp, is_ua_dealloc == 1),
    UAAllocTime = minif(Timestamp, is_ua_alloc == 1),
    EndTime = max(Timestamp)
  by HostName, big_group_id, sub_group_id, Endpoint, ClusterName
| where isnotnull(StartTime) and isnotnull(UADeallocTime)
| extend
    UA_Time = todouble(UADeallocTime - StartTime)/todouble(1h),
    Allocation_Time = todouble(UAAllocTime - UADeallocTime)/todouble(1h),
    Validation_Time = todouble(EndTime - UAAllocTime)/todouble(1h),
    UA_cycle_Time = todouble(EndTime - StartTime)/todouble(1h)
| sort by StartTime
| extend NodeRecycleEventID = strcat(HostName, '_', big_group_id)
| project-away UADeallocTime, UAAllocTime;
//Step 4: Aggregates Metrics Per UA Event
let event_summary_metrics =
subgroup_metrics
| summarize
    EventStartTime = min(StartTime),
    EventEndTime = max(EndTime),
    NodeRecycleTime_event = sum(UA_cycle_Time),
    CountUAAction = countif(isnotnull(UA_cycle_Time))
by NodeRecycleEventID
| extend Week = weekofyear(startofweek(EventStartTime - 1d, 0));// Monday as the start of the week
//Step 5: Joins and Summarizes Data to Create Weekly Metrics of UACycleTime, UATime, AllocationTime, ValidationTime for each ClusterName and Endpoint
let metrics_with_event_summary =
subgroup_metrics
| join kind=leftouter (
    event_summary_metrics
) on NodeRecycleEventID;
let WeeklySubGroupMetrics =
metrics_with_event_summary
| summarize
    UACycleTime_weekly = avg(UA_cycle_Time),
    UATime_weekly = avg(UA_Time),
    AllocationTime_weekly = avg(Allocation_Time),
    ValidationTime_weekly = avg(Validation_Time)
by Week, ClusterName, Endpoint;
//Step 6: Joins and Summarizes Data to Create Weekly Metrics of OFR NodeCount, OFR NodeRecycleTime, Single Recycel UA Time for each ClusterName and Endpoint
//This step is necessary to calculate the Node Recycle Time, as each node may experience
let WeeklyNodeRecycleEventMetrics =
metrics_with_event_summary
| summarize take_any(NodeRecycleTime_event), take_any(CountUAAction) by NodeRecycleEventID, Week, ClusterName, Endpoint
| summarize
    NodeCount = dcount(NodeRecycleEventID), // correct
    NodeRecycleTime_weekly = avg(NodeRecycleTime_event),
    UATimeinSingleRecycel_weekly = avg(CountUAAction)
by Week, ClusterName, Endpoint;
//Step 7: Join to Get Final Result
let Res =
WeeklySubGroupMetrics
| join kind=inner WeeklyNodeRecycleEventMetrics on Week, ClusterName, Endpoint
| order by Week asc;
Res
```

grouped_by_week

## Example 10: Weekly Failure Rate, for each Endpoint and Failure Category
Question: "What is the weekly failure ratio grouped by each failure category and endpoint?"

Generated KQL:
```kql
// Step 0, take the original data from source
let src =
cluster('{cluster_url}')
    .database('{database_name}')
    .table('NodeActionRecord')
| join kind=leftouter NodeActionAttributes on Action
| sort by HostName asc, Timestamp asc, Action asc
| serialize;
// Step 1, Mark every row that has a non-empty Phase
let withMark =
    src
    | extend NonEmptyFlag = iff(isnotempty(Phase), 1, 0);
// Step 2, Build a running total of NonEmptyFlag per HostName
let running =
    withMark
    | extend PhaseGroup =
        row_cumsum(NonEmptyFlag)   // needs `serialize` earlier
        ;   // PhaseGroup increments each time we see a non-empty Phase
// Step 3, Look up the _latest_ non-empty Phase in each PhaseGroup
let phaseLookup =
    running
    | where NonEmptyFlag == 1
    | summarize LastPhase = arg_max(Timestamp, Phase) by HostName, PhaseGroup
    | project HostName, PhaseGroup, LastPhase = Phase;
// Step 4, Join back to fill down the LastPhase
let enriched = running
| join kind=inner (phaseLookup) on HostName, PhaseGroup
| project-away NonEmptyFlag, Action1, HostName1, PhaseGroup1
| sort by HostName asc, Timestamp asc;
// Step 5: Earliest timestamp per HostName + PhaseGroup
let groupStart =
    enriched
    | summarize StartTime = min(Timestamp) by HostName, PhaseGroup;
// Step 6: Stitch them together and filter for relevant records
let record = enriched
| join kind=inner groupStart on HostName, PhaseGroup
| project-reorder StartTime, Timestamp, HostName, PhaseGroup
| sort by Timestamp asc
| where Action contains "cordoned-triaged" and LastPhase contains "routine"
| summarize arg_max(Timestamp, *) by HostName, PhaseGroup // Get the latest record per HostName, PhaseGroup
| project-away HostName1, PhaseGroup1, Phase // Clean up redundant columns
| extend CategoryReason = strcat(Category, "-", Reason); // Create CategoryReason
// Step 7, Define date range
let startDate = datetime(2025-03-17);
let endDate = now();
let allDates = range StartDate from startDate to endDate step 7d;
// Step 8, Get all distinct Endpoints from the 'record' table
let allEndpoints = record | distinct Endpoint;
// Step 9, Get all distinct Categories from the 'record' table
let allCategories = record | distinct Category;
// Step 10, Create a Cartesian product of all dates, endpoints, and categories, this ensures every combination exists as a row
let allCombinations =
    allDates
    | extend dummy = 1 // Add a dummy column for cross-joining
    | join kind=inner (allEndpoints | extend dummy = 1) on dummy
    | join kind=inner (allCategories | extend dummy = 1) on dummy
    | project StartDate, Endpoint, Category; // Select the desired combination columns
// Step 11, Summarize the actual data from 'record' for joining
let summarizedRecord =
    record
    | summarize Total = count() by bin(StartTime, 7d), Endpoint, Category;
// Step 12, Left outer join the full combinations with the summarized data, this will keep all combinations and fill 'Total' with 0 where no match is found
let full = allCombinations
| join kind=leftouter summarizedRecord on $left.StartDate == $right.StartTime, Endpoint, Category
| project StartDate,
          Endpoint,
          Category,
          WeeklyTotal = coalesce(Total, 0) // Replace null Total with 0
| extend CurrentWeekStart = startofweek(now(), 0) + 1d // Calculate the start of the current week (Monday)
| extend DaysInCurrentWeek = datetime_diff('day', now(), CurrentWeekStart) + 1 // Calculate days passed in current week
| extend DaysinWeek = iff(StartDate == CurrentWeekStart, DaysInCurrentWeek, 7)
| extend Date = StartDate
| extend Week = weekofyear(startofweek(StartDate - 1d, 0)) // Monday as the start of the week
| extend Category = iff(isempty(Category), "uncategorized", Category) // Handle potentially empty categories
| order by StartDate asc, Endpoint asc, Category asc
| project-away StartDate, CurrentWeekStart, DaysInCurrentWeek;
// Step 13, Calculate the weekly ratio
let result = full
| extend AvgFailureRatio = iff(DaysinWeek == 0, 0.0, todouble(WeeklyTotal) / DaysinWeek)
| project Week, Date, Endpoint, Category, AvgFailureRatio;
result
```
